# requires
```
cat > ~/.aws/credentials << EOF
[default]
aws_access_key_id='some_id'
aws_secret_access_key='some_key'
EOF
```

ssh-keygen

eval $(ssh-agent)
ssh-add ~/.ssh/id_rsa_aws_1
